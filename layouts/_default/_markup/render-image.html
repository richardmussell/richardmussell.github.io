{{- /* Render Hook for Images - Performance Optimized with WebP Support - WCAG 1.1.1 */ -}} {{- $imagePath := .Destination | safeURL -}} {{- $alt := .Text -}} {{- $title := .Title | default "" -}} {{- /* If no alt text provided, use title or mark as decorative */ -}} {{- if not $alt -}} {{- if $title -}} {{- $alt = $title -}} {{- else -}} {{- /* Mark as decorative if no alt or title provided */ -}} {{- $alt = "" -}} {{- end -}} {{- end -}} {{- /* Get image resource for processing */ -}} {{- $image := resources.Get $imagePath -}} {{- $isLazy := true -}} {{- $width := 0 -}} {{- $height := 0 -}} {{- /* Check if image is above the fold (first image in content) */ -}} {{- if eq .Ordinal 1 -}} {{- $isLazy = false -}} {{- end -}} {{- if $image -}} {{- /* Process image with Hugo Pipes - Performance Optimized */ -}} {{- $width = $image.Width -}} {{- $height = $image.Height -}} {{- /* Generate responsive srcset: Mobile (500w), Tablet (800w), Desktop (1200w) */ -}} {{- $srcset := slice -}} {{- $srcsetWebp := slice -}} {{- $srcsetAvif := slice -}} {{- $sizes := (slice 500 800 1200) -}} {{- range $sizes -}} {{- if le . $image.Width -}} {{- /* Resize image */ -}} {{- $resized := $image.Resize (printf "%dx" .) -}} {{- /* Generate WebP version */ -}} {{- $resizedWebp := $resized | resources.Filter "webp" -}} {{- /* Generate AVIF version (next-gen, smaller than WebP) */ -}} {{- $resizedAvif := $resized | resources.Filter "avif" -}} {{- if $resized -}} {{- $srcset = $srcset | append (printf "%s %dw" $resized.RelPermalink .) -}} {{- end -}} {{- if $resizedWebp -}} {{- $srcsetWebp = $srcsetWebp | append (printf "%s %dw" $resizedWebp.RelPermalink .) -}} {{- end -}} {{- if $resizedAvif -}} {{- $srcsetAvif = $srcsetAvif | append (printf "%s %dw" $resizedAvif.RelPermalink .) -}} {{- end -}} {{- end -}} {{- end -}} {{- /* Generate fallback versions (original size) */ -}} {{- $webp := $image | resources.Filter "webp" -}} {{- $avif := $image | resources.Filter "avif" -}} {{- else -}} {{- /* Fallback for static images not in resources */ -}} {{- end -}}
